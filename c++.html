    أهم المفاهيم والأمثلة العملية لتعلم C++ بسرعة

    مقدمة

    C++ هي لغة برمجة قوية وموجهة للكائنات، تُستخدم لتطوير الألعاب، التطبيقات، والبرمجيات عالية الأداء. سنتعرف على أساسيات اللغة مع أمثلة عملية.

    1️⃣ أول برنامج C++

    #include <iostream>
    using namespace std;
    int main() {
      cout << "مرحباً بالعالم!" << endl;
      return 0;
    }

    البرنامج الأساسي لطباعة نص على الشاشة.

    2️⃣ المتغيرات والأنواع

    int age = 25; // عدد صحيح double height = 1.75; // رقم عشري char grade = 'A'; // حرف bool isStudent = true; // بوليان

    3️⃣ العمليات الحسابية

    int sum = 5 + 3; int diff = 10 - 4; int mult = 6 * 2; double div = 8.0 / 2.0; int mod = 7 % 3; int exp = pow(2,3); // 8

    4️⃣ الشروط

    if (age >= 18) {
      cout << "بالغ" << endl;
    } else {
      cout << "قاصر" << endl;
    }

    5️⃣ الحلقات

    for(int i=0;i<5;i++){
      cout << i << endl;
    }

    حلقة for من 0 إلى 4.

    int i = 0;
    while(i<5){
      cout << i << endl;
      i++;
    }

    حلقة while.

    6️⃣ الدوال (Functions)

    int sum(int a,int b){
      return a+b;
    }
    cout << sum(5,3) << endl; // 8

    7️⃣ المصفوفات (Arrays)

    int numbers[5] = {1,2,3,4,5}; cout << numbers[0] << endl; // 1

    8️⃣ الكائنات (Objects)

    class Person {
      public:
      string name;
      int age;
    };
    Person p;
    p.name = "أحمد";
    p.age = 25;

    9️⃣ التعليقات

    // تعليق على سطر واحد /* تعليق على أكثر من سطر */

    cout << "مرحباً" << endl; int num;
